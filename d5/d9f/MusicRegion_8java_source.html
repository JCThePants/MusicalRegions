264  @Override
265  protected void onPlayerEnter(Player player, EnterRegionReason reason) {
266 
267  if (isExcluded(player) || _playList == null)
268  return;
269 
270  PlayList playList = _playLists.get(player.getUniqueId());
271  if (playList != null) {
272 
273  // don't play playlist if player is already listening to a queue
274  PlayerSoundQueue queue = playList.getSoundQueue(player);
275  if (queue != null)
276  return;
277  }
278 
279  RegionPlayEvent event = new RegionPlayEvent(player, this, _playList);
280  Nucleus.getEventManager().callBukkit(this, event);
281  if (event.isCancelled())
282  return;
283 
284  PlayerSoundQueue queue = event.getPlayList().addPlayer(player, _settings);
285  if (queue != null) {
286  queue.getMeta().setKey(META_KEY, this);
287 
288  _playLists.put(player.getUniqueId(), event.getPlayList());
289  }
290  }
